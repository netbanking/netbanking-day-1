CREATE TABLE LOGIN(
LOGIN_ID VARCHAR(25) NOT NULL,
PASSWORD VARCHAR(10) NOT NULL,
ROLE VARCHAR (10) NOT NULL
CONSTRAINT PK_LOGIN_ID PRIMARY KEY(LOGIN_ID),
);

CREATE TABLE CUSTOMER (
CUSTOMER_ID NUMBER(10) GENERATED BY DEFAULT AS IDENTITY,
FIRST_NAME VARCHAR(50) NOT NULL,
LAST_NAME VARCHAR(25) NOT NULL,
LOGIN_ID VARCHAR(25) NOT NULL,
ADDRESS VARCHAR(50) NOT NULL,
CITY VARCHAR(20) NOT NULL,
STATE VARCHAR (20) NOT NULL,
ZIP NUMBER(6) NOT NULL,
PHONE_NUMBER NUMBER(10) NOT NULL,
TELE_PHONE_NUMBER NUMBER(8) NOT NULL,
EMAIL_ID VARCHAR(30) NOT NULL,
STATUS VARCHAR(20) DEFAULT 'NEW' NOT NULL,
CONSTRAINT PK_CUSTOMER_ID PRIMARY KEY(CUSTOMER_ID),
CONSTRAINT FK_LOGIN_ID FOREIGN KEY (LOGIN_ID)
REFERENCES LOGIN(LOGIN_ID),
CONSTRAINT CK_STATUS CHECK (STATUS IN ('APPROVED','REJECTED')) 
);

SELECT * FROM customer;

CREATE TABLE ACCOUNT(
ACCOUNT_ID NUMBER(10) GENERATED BY DEFAULT AS IDENTITY,
ACCOUNT_TYPE VARCHAR(20) NOT NULL,
OPEN_DATE DATE NOT NULL,
MINIMUM_BALANCE NUMBER(7,2)NOT NULL,
CURRENT_BALANCE NUMBER(7,2) NOT NULL,
RATE_OF_INTEREST NUMBER(3,2) NOT NULL,
CUSTOMER_ID NUMBER(10) NOT NULL,
CONSTRAINT PK_ACCOUNT_ID PRIMARY KEY (ACCOUNT_ID),
CONSTRAINT FK_CUSTOMER_ID FOREIGN KEY (CUSTOMER_ID)
REFERENCES CUSTOMER(CUSTOMER_ID)
);

SELECT * FROM account;
